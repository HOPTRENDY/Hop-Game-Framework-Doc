C++ Usage: SetByCaller Magnitudes

// Create effect spec
FGameplayEffectSpecHandle SpecHandle = ASC->MakeOutgoingSpec(
    GE_MeleeDamage, 
    1, // Level
    ASC->MakeEffectContext()
);

// Set magnitude by Data tag
if (SpecHandle.IsValid())
{
    FGameplayEffectSpec* Spec = SpecHandle.Data.Get();
    
    // Set damage
    FGameplayTag DamageTag = FHopComboGraphGameplayTags::Get().Data_Damage;
    Spec->SetSetByCallerMagnitude(DamageTag, FinalDamage);
    
    // Set combo multiplier
    FGameplayTag ComboTag = FHopComboGraphGameplayTags::Get().Data_ComboMultiplier;
    float ComboMultiplier = 1.0f + (ComboCount * 0.1f); // +10% per hit
    Spec->SetSetByCallerMagnitude(ComboTag, ComboMultiplier);
    
    // Apply effect
    ASC->ApplyGameplayEffectSpecToSelf(*Spec);
}

Example 2: Combo Damage Scaling with Tags

Scenario: Apply increasing damage based on combo hit count.

C++ Implementation

// In GA_MeleeComboAttack::ApplyDamage()
void UGA_MeleeComboAttack::ApplyDamage(AActor* Target, int32 ComboCount)
{
    // Base damage
    float BaseDamage = 50.0f;
    
    // Combo scaling: +10% per hit
    float ComboMultiplier = 1.0f + (ComboCount * 0.1f);
    float FinalDamage = BaseDamage * ComboMultiplier;
    
    // Get appropriate combo tag
    FGameplayTag DamageTag;
    switch(ComboCount)
    {
        case 1: DamageTag = FHopGasGameplayTags::Get().Damage_Physical_Slash_Combo1; break;
        case 2: DamageTag = FHopGasGameplayTags::Get().Damage_Physical_Slash_Combo2; break;
        case 3: DamageTag = FHopGasGameplayTags::Get().Damage_Physical_Slash_Combo3; break;
        case 4: DamageTag = FHopGasGameplayTags::Get().Damage_Physical_Slash_Combo4; break;
    }
    
    // Create effect spec
    FGameplayEffectSpecHandle SpecHandle = MakeOutgoingGameplayEffectSpec(
        GE_MeleeDamage,
        GetAbilityLevel()
    );
    
    if (SpecHandle.IsValid())
    {
        // Set damage magnitude
        SpecHandle.Data->SetSetByCallerMagnitude(
            FHopComboGraphGameplayTags::Get().Data_Damage,
            FinalDamage
        );
        
        // Add combo damage tag
        SpecHandle.Data->DynamicAssetTags.AddTag(DamageTag);
        
        // Apply to target
        ApplyGameplayEffectSpecToTarget(
            CurrentSpecHandle,
            CurrentActorInfo,
            CurrentActivationInfo,
            SpecHandle,
            GetTargetDataFromActivationInfo()
        );
    }
}

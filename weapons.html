Example 3: Weapon Pickup & Equip (C++)

// In character class
void AMyCharacter::PickupWeapon(AHopWeaponBase* Weapon)
{
    if (!Weapon || !Weapon->CanBePickedUp()) return;

    // Try pickup via interaction interface
    if (Weapon->TryPickup(this))
    {
        // Weapon automatically grants abilities in OnWeaponEquipped
        // Store reference
        CurrentWeapon = Weapon;
        
        // Attach to socket
        Weapon->AttachToComponent(
            GetMesh(),
            FAttachmentTransformRules::SnapToTargetIncludingScale,
            Weapon->EquipSocket
        );
    }
}

void AMyCharacter::DropWeapon()
{
    if (!CurrentWeapon) return;

    // Set weapon to dropped state (removes abilities)
    CurrentWeapon->SetWeaponState(EHopWeaponState::Dropped);
    
    // Detach from character
    CurrentWeapon->DetachFromActor(FDetachmentTransformRules::KeepWorldTransform);
    
    // Prepare for pickup
    CurrentWeapon->PrepareForPickup();
    
    CurrentWeapon = nullptr;
}

‚úÖ Best Practices

  • ‚úÖ Always use SetWeaponState() for state changes (auto-manages abilities)
  • ‚úÖ Grant abilities via WeaponAbilities array (automatic on equip)
  • ‚úÖ Use OverlayAnimLayer for weapon-specific animations
  • ‚úÖ Attach weapons to proper sockets (hand for equipped, back/hip for holstered)
  • ‚úÖ Use Gameplay Effects for weapon stats (BuffDamage, BuffAccuracy)
  • ‚úÖ Leverage WeaponTags for filtering (Weapon.Ranged, Weapon.TwoHanded)
  • ‚ö†Ô∏è Don't manually grant abilities - let weapon system handle it
  • ‚ö†Ô∏è Don't forget to set DamageMode and FireMode (required for combat logic)

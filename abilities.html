C++ Example

void AMyCharacter::InitializeAbilities()
{
    if (!AbilitySystemComponent) return;
    
    FGameplayAbilitySpec MoveSpec(UGA_Move::StaticClass(), 1);
    MoveSpec.InputID = -1;
    AbilitySystemComponent->GiveAbilityAndActivateOnce(MoveSpec);
    
    FGameplayAbilitySpec SprintSpec(UGA_Sprint::StaticClass(), 1);
    SprintSpec.InputID = static_cast<int32>(EAbilityInputID::Sprint);
    AbilitySystemComponent->GiveAbility(SprintSpec);
}

üîß Integration Guide

‚ö†Ô∏è Prerequisites: Ensure your character has an AbilitySystemComponent and CombatComponent before integration.

C++ Character Setup

C++ Header (.h) 
// MyCharacter.h
#pragma once

#include "CoreMinimal.h"
#include "GameFramework/Character.h"
#include "AbilitySystemInterface.h"
#include "MyCharacter.generated.h"

UCLASS()
class YOURPROJECT_API AMyCharacter : public ACharacter, public IAbilitySystemInterface
{
    GENERATED_BODY()

public:
    AMyCharacter();

    // IAbilitySystemInterface
    virtual UAbilitySystemComponent* GetAbilitySystemComponent() const override;

protected:
    // Components
    UPROPERTY(VisibleAnywhere, BlueprintReadOnly, Category = "Abilities")
    TObjectPtr<UAbilitySystemComponent> AbilitySystemComponent;

    UPROPERTY(VisibleAnywhere, BlueprintReadOnly, Category = "Combat")
    TObjectPtr<UCombatComponent> CombatComponent;

    // Attribute Set
    UPROPERTY()
    TObjectPtr<UHopCombatAttributeSet> AttributeSet;

    // Abilities to grant
    UPROPERTY(EditDefaultsOnly, BlueprintReadOnly, Category = "Abilities")
    TArray<TSubclassOf<UGameplayAbility>> DefaultAbilities;

    virtual void BeginPlay() override;
    void InitializeAbilities();
};
C++ Implementation (.cpp) 
// MyCharacter.cpp
#include "MyCharacter.h"
#include "AbilitySystemComponent.h"
#include "Components/CombatComponent.h"
#include "Attributes/HopCombatAttributeSet.h"

AMyCharacter::AMyCharacter()
{
    // Create Ability System Component
    AbilitySystemComponent = CreateDefaultSubobject<UAbilitySystemComponent>(TEXT("AbilitySystemComponent"));
    AbilitySystemComponent->SetIsReplicated(true);
    AbilitySystemComponent->SetReplicationMode(EGameplayEffectReplicationMode::Mixed);

    // Create Combat Component
    CombatComponent = CreateDefaultSubobject<UCombatComponent>(TEXT("CombatComponent"));
    CombatComponent->SetIsReplicated(true);

    // Create Attribute Set
    AttributeSet = CreateDefaultSubobject<UHopCombatAttributeSet>(TEXT("AttributeSet"));
}

UAbilitySystemComponent* AMyCharacter::GetAbilitySystemComponent() const
{
    return AbilitySystemComponent;
}

void AMyCharacter::BeginPlay()
{
    Super::BeginPlay();

    if (HasAuthority() && AbilitySystemComponent)
    {
        // Initialize attribute set
        AttributeSet->InitHealth(100.0f);
        AttributeSet->InitMaxHealth(100.0f);

        // Grant abilities
        InitializeAbilities();
    }
}

void AMyCharacter::InitializeAbilities()
{
    if (!HasAuthority() || !AbilitySystemComponent)
    {
        return;
    }

    for (TSubclassOf<UGameplayAbility>& AbilityClass : DefaultAbilities)
    {
        if (AbilityClass)
        {
            AbilitySystemComponent->GiveAbility(
                FGameplayAbilitySpec(AbilityClass, 1, INDEX_NONE, this)
            );
        }
    }
}

The HOP Framework is designed with multiplayer in mind. Follow these guidelines to ensure smooth networked gameplay.

üéÆ Interaction System (NEW)

Multiplayer-Safe

‚úÖ Use These Helper Functions:

  • Should Show Interaction UI() - Returns true ONLY for client players (NOT listen server!)
  • Get Current Interactable For UI() - Safe getter (null on server/other clients)
  • Has Interactable Target() - Multiplayer-safe existence check

‚ùå DON'T Do This:

// BAD - Will crash on server, show on listen server
OnInteractableTargeted Event
  ‚Üí Create Widget
  ‚Üí Add to Viewport  // Unwanted on listen server!

‚úÖ DO This Instead:

// GOOD - Only shows on clients
OnInteractableTargeted Event
  ‚Üí Should Show Interaction UI
     ‚Üí TRUE: Create Widget (clients only!)
     ‚Üí FALSE: Do Nothing (listen server skipped)

Important: ShouldShowInteractionUI() excludes the listen server player to prevent UI clutter on the server viewport.
